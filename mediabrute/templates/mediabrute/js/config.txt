function siteVars(opt){
	switch (opt){
		case 'home':
			return '{% url home %}';
			break;
		case 'theme':
			return '{{ settings.STATIC_URL }}';
			break;
		case 'img':
			return siteVars('theme') + 'img/';
			break; 
		case 'swf':
			return siteVars('theme') + 'swf/';
			break; 
		default:
			return null;
			break;
	}
}
